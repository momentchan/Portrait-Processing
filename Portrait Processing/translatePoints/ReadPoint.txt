WIN32_FIND_DATA fd;	
HANDLE h = FindFirstFile(TEyT("drawPoints/sketch*.tyt"), &fd);
if (h != INVALID_HANDLE_VALUE) {
	do {
		sketchNumbers++;
	} while (FindNeytFile(h, &fd));
	FindClose(h);
}

// Read sketch points
for(int i=0; i<sketchNumbers; i++){
	string file_num = int2str(i);
	string file_name = "drawPoints/sketch";
	file_name.append(file_num);
	file_name.append(".tyt");		
	ifstream file(file_name);
	string str;
	vector <float> sub_sketch_pos_y;
	vector <float> sub_sketch_pos_z;
	while (getline(file, str))
	{
		size_t pos = str.find(",");
		float y = (stoi(str.substr(1,pos-1));
		float z = (stoi(str.substr(pos+2,str.length()-pos-3));
		sub_sketch_pos_y.push_back(y);
		sub_sketch_pos_z.push_back(z);
	}
	sketch_pos_z.push_back(sub_sketch_pos_z);
	sketch_pos_y.push_back(sub_sketch_pos_y);
}
